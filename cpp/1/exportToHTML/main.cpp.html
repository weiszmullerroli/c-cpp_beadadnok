#include <iostream>
#include <bits/stdc++.h>
using namespace std;

double sqrt_newton(double num, double x0){
    double x = num;
    double sol;
    int i =1 ;
    while (i == 1){
        sol= (x+(num/x))/2 ;
        if (abs(sol-x)<x0){
            i = 0;
        }
        x=sol;
    }
    return sol;
}

int main(){
    cout << sqrt_newton(480,0.0001);
    return 0;
}
